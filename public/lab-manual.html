4b User Authentication (JWT)
// auth.js — install: npm install express jsonwebtoken bcryptjs
const express = require("express");
const jwt = require("jsonwebtoken");
const bcrypt = require("bcryptjs");
const app = express();
app.use(express.json());

const SECRET = "skav_jwt_secret";
let users = [];

app.post("/register", async (req, res) => {
  const hash = await bcrypt.hash(req.body.password, 10);
  users.push({ username: req.body.username, password: hash });
  res.json({ message: "Registered successfully" });
});

app.post("/login", async (req, res) => {
  const user = users.find(u => u.username === req.body.username);
  if (!user || !(await bcrypt.compare(req.body.password, user.password)))
    return res.status(401).json({ error: "Invalid credentials" });
  const token = jwt.sign({ username: user.username }, SECRET, { expiresIn: "1h" });
  res.json({ token });
});

const verifyToken = (req, res, next) => {
  const token = req.headers.authorization?.split(" ")[1];
  if (!token) return res.status(403).json({ error: "No token" });
  try {
    req.user = jwt.verify(token, SECRET);
    next();
  } catch { res.status(401).json({ error: "Invalid token" }); }
};

app.get("/dashboard", verifyToken, (req, res) =>
  res.json({ message: `Welcome, ${req.user.username}!` })
);

app.listen(3000);
Expected Output
POST /register {"username":"SKAV","password":"pass123"} → {"message":"Registered successfully"} POST /login {"username":"SKAV","password":"pass123"} → {"token":"eyJhbGciOi..."} GET /dashboard (Authorization: Bearer <token>) → {"message":"Welcome, SKAV!"}

// Events.jsx
import { useState } from "react";

export default function Events() {
  const [text, setText] = useState("");
  const [log, setLog] = useState([]);

  const handleClick = () =>
    setLog(prev => [...prev, `Clicked at ${new Date().toLocaleTimeString()}`]);

  const handleChange = (e) => setText(e.target.value);

  const handleKeyPress = (e) => {
    if (e.key === "Enter")
      setLog(prev => [...prev, `Submitted: ${text}`]);
  };

  return (
    <div>
      <input onChange={handleChange} onKeyDown={handleKeyPress} placeholder="Type and press Enter" />
      <button onClick={handleClick}>Click Me</button>
      <ul>{log.map((l, i) => <li key={i}>{l}</li>)}</ul>
    </div>
  );
}

// Forms.jsx
import { useState } from "react";

export default function RegistrationForm() {
  const [form, setForm] = useState({
    name: "", email: "", gender: "male",
    course: "React", agree: false
  });
  const [submitted, setSubmitted] = useState(null);

      <input name="name" value={form.name} onChange={handle} placeholder="Name" />
      <input name="email" value={form.email} onChange={handle} placeholder="Email" />

        <option>Node.js</option>
        <option>TypeScript</option>
      </select>
      <label>
        <input type="checkbox" name="agree" checked={form.agree} onChange={handle} /> I agree
      </label>
      <button onClick={() => setSubmitted(form)}>Submit</button>
      {submitted && <pre>{JSON.stringify(submitted, null, 2)}</pre>}
    </div>

// Hooks.jsx
import { useState, useEffect, useRef, useCallback } from "react";

export default function HooksDemo() {
  // useState
  const [count, setCount] = useState(0);

  // useRef — persists across renders, no re-render
  const inputRef = useRef(null);
  const renderCount = useRef(0);

  // useEffect — runs after render
  useEffect(() => {
    renderCount.current++;
    document.title = `Count: ${count}`;
  }, [count]);

  // useCallback — memoize functions
  const increment = useCallback(() => setCount(c => c + 1), []);

  return (
    <div>
      <p>Count: {count}</p>
      <p>Renders triggered: {renderCount.current}</p>
      <button onClick={increment}>Increment (useCallback)</button>
      <input ref={inputRef} placeholder="useRef input" />
      <button onClick={() => inputRef.current.focus()}>Focus Input</button>
    </div>
  );
}

// Context.jsx
import { createContext, useContext, useState } from "react";

const ThemeContext = createContext(null);

function ThemeProvider({ children }) {
  const [theme, setTheme] = useState("dark");
  const toggle = () => setTheme(t => t === "dark" ? "light" : "dark");
  return (
    <ThemeContext.Provider value={{ theme, toggle }}>
      {children}
    </ThemeContext.Provider>
  );
}

function Header() {
  const { theme, toggle } = useContext(ThemeContext);
  return <button onClick={toggle}>Current theme: {theme}</button>;
}

function Page() {
  const { theme } = useContext(ThemeContext);
  return <p>Page knows theme is: {theme}</p>;
}

export default function App() {
  return (
    <ThemeProvider>
      <Header />
      <Page />

  );

// TodoApp.jsx
import { useState } from "react";

export default function TodoApp() {
  const [todos, setTodos] = useState([
    { id: 1, text: "Learn TypeScript", done: true },
    { id: 2, text: "Build Togetherly v2", done: false }
  ]);
  const [input, setInput] = useState("");

  const add = () => {
    if (!input.trim()) return;
    setTodos([...todos, { id: Date.now(), text: input, done: false }]);
    setInput("");
  };

  const toggle = (id) =>
    setTodos(todos.map(t => t.id === id ? { ...t, done: !t.done } : t));

  const remove = (id) =>
    setTodos(todos.filter(t => t.id !== id));

  return (
    <div>
      <h2>My Tasks ({todos.filter(t => !t.done).length} remaining)</h2>
      <input value={input} onChange={e => setInput(e.target.value)} onKeyDown={e => e.key === "Enter" && add()} />
      <button onClick={add}>Add</button>
      <ul>
        {todos.map(t => (
          <li key={t.id} style={{ textDecoration: t.done ? "line-through" : "none" }}>
            <input type="checkbox" checked={t.done} onChange={() => toggle(t.id)} />
            {t.text}
            <button onClick={() => remove(t.id)}>Delete</button>
          </li>
        ))}
      </ul>
    </div>
  );
}

// server.js — Full To-do App
const express = require("express");
const path = require("path");
const app = express();

app.use(express.json());
app.use(express.urlencoded({ extended: true }));
app.use(express.static("public"));
app.set("view engine", "ejs");

let todos = [];
let idCounter = 1;

// View all todos
app.get("/", (req, res) => {
  res.render("index", { todos, filter: req.query.filter || "all" });
});

// Add todo
app.post("/add", (req, res) => {
  todos.push({ id: idCounter++, text: req.body.text, done: false });
  res.redirect("/");
});

// Toggle done
app.post("/toggle/:id", (req, res) => {
  const t = todos.find(t => t.id === parseInt(req.params.id));
  if (t) t.done = !t.done;
  res.redirect("/");
});

// Delete todo
app.post("/delete/:id", (req, res) => {
  todos = todos.filter(t => t.id !== parseInt(req.params.id));
  res.redirect("/");
});

// REST API endpoints
app.get("/api/todos", (req, res) => res.json(todos));
app.post("/api/todos", (req, res) => {
  const t = { id: idCounter++, text: req.body.text, done: false };
  todos.push(t);
  res.status(201).json(t);
});

app.listen(3000, () => console.log("Todo App running on http://localhost:3000"));
